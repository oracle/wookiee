34 /**
35  * This class manages the creation of both the KafkaConsumerCoordinator (if 'consumer' is configured)
36  * and the KafkaWriter (if 'producer' is configured). One can retrieve either of these actors by
37  * making a call to GetCoordinator or GetWriters
38  */
39 object KafkaManager {
40   // User this to retrieve the coordinator in charge of managing Workers
41   case object GetCoordinator
42   //Will return the distributor
43   case object GetDistributor
44   // Kafka producer started up if 'producer' is configured
45   var producer: Option[ActorRef] = None
46 }

48 class KafkaManager(name: String) extends Component(name) with KafkaSettings {
49   import KafkaManager._
50   import context.dispatcher
51   // Consumer coordinator started up if 'consumer' is configured
52   var coordinator: Option[ActorRef] = None
53 
54   // Source monitor started if source-monitor = true
55   var sourceMonitor: Option[ActorRef] = None
56 
57   // Consumer proxy, started up if 'consumer' is configured. Used to retrieve topic
58   // information and the data to consume
59   var consumerManager: Option[ActorRef] = None
60 
61   //Distributor actor will start up if a 'consumer' is configured
62   var distributor: Option[ActorRef] = None
63 
64   var consumerManagerHealth: Option[HealthComponent] = None
65 
66   // Start up the producer as soon as we can since it has no dependencies
67   override def start = {
68     if (kafkaConfig.hasPath("producer")) {
69       startProducer()
70     }
71     super.start
72   }
73 
74   override def receive = super.receive orElse configReceive orElse {
75     // Use this call to get the Kafka Consumer Coordinator, if configured
76     case GetCoordinator =>
77       sender ! coordinator
78 
79     // We are loading the coordinator here because it depends on zookeeper component being loaded up which
80     // will have occurred earlier at the ComponentStart phase
81     case SystemReady =>
82       if (kafkaConfig.hasPath("consumer")) {
83         startCoordinator()
84       }
85 
86     case GetDistributor =>
87       sender ! distributor
88 
89     case PrepareForShutdown =>
90       coordinator.foreach(_ ! PrepareForShutdown)
91       distributor.foreach(_ ! PrepareForShutdown)
92 
93     case h: HealthComponent =>
94       consumerManagerHealth = Some(h)
95   }
96 
97   def startProducer() {
98     log.info("Starting producer as wookiee-kafka config contained 'producer' config")
99     producer = Some(context.system.actorOf(Props[KafkaWriter], "producer"))
100   }
101 
102   override def checkHealth: Future[HealthComponent] = {
103     val p = Promise[HealthComponent]()
104     val children = Seq(distributor, coordinator, producer)
105 
106     getHealth.onComplete {
107       case Success(s) =>
108         val healthFutures = children.flatten map { ref =>
109           (ref ? CheckHealth).mapTo[HealthComponent] recover {
110             case ex: Exception => HealthComponent(ref.path.name, ComponentState.CRITICAL, s"Failure to get health of child component. ${ex.getMessage}")
111           }
112         }
113 
114         Future.sequence(healthFutures) onComplete {
115           case Failure(f) =>
116             log.debug(f, "Failed to retrieve health of children objects")
117             p success HealthComponent(s.name, ComponentState.CRITICAL, s"Failure to get health of child components. ${f.getMessage}")
118           case Success(healths) =>
119             healths foreach { it => s.addComponent(it) }
120             p success s
121         }
122       case Failure(f) =>
123         log.debug(f, "Failed to get health from component")
124         p success HealthComponent(self.path.toString, ComponentState.CRITICAL, f.getMessage)
125     }
126 
127     p.future
128   }
129 
130   override protected def getHealth: Future[HealthComponent] = {
131     val h = HealthComponent(self.path.toString, ComponentState.NORMAL, "Healthy")
132     consumerManagerHealth.foreach(h.addComponent)
133     Future.successful(h)
134   }
135 
136   /**
137    * Start up all necessary actors
138    */
139   def startCoordinator() {
140     if(coordinator.isEmpty) {
141       log.info(s"Starting coordinator class")
142       if (Try { kafkaConfig.getBoolean("monitor-sources") } getOrElse false) {
143         log.info("'monitor-sources' is true, starting Source Monitor")
144         sourceMonitor = Some(context.actorOf(Props(classOf[SourceMonitor]), "source-monitor"))
145       }
146       consumerManager = Some(context.actorOf(KafkaTopicManager.props(sourceMonitor), "consumer-manager"))
147       coordinator = Some(context.actorOf(Props(leader, consumerManager.get, sourceMonitor), "consumer-coordinator"))
148       distributor = Some(context.actorOf(KafkaConsumerDistributor.props(consumerManager.get), "consumer-distributor"))
149     }
150   }
151 
152   override def renewConfiguration() {
153     log.info("Received config change message, checking hosts for changes...")
154     super.renewConfiguration()
155     if (coordinator.isDefined) {
156       coordinator.get ! ConfigChange()
157     }
158   }
159 
160   override def postStop() {
161     stop
162   }
163 
164   // Stops the coordinator, unregistering this node from zookeeper
165   override def stop {
166     context.children foreach { child â‡’
167       log.info(s"Stopping child [${child.path}]")
168       context.stop(child)
169     }
170     coordinator = None
171     consumerManager = None
172     distributor = None
173     producer = None
174   }
175 }
