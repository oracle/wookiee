34 /**
35  * This class manages the creation of both the KafkaConsumerCoordinator (if 'consumer' is configured)
36  * and the KafkaWriter (if 'producer' is configured). One can retrieve either of these actors by
37  * making a call to GetCoordinator or GetWriters
38  */

41   case object GetCoordinator
42   //Will return the distributor
43   case object GetDistributor
44   // Kafka producer started up if 'producer' is configured
45   var producer: Option[ActorRef] = None

48 class KafkaManager(name: String) extends Component(name) with KafkaSettings {
49   import KafkaManager._
50   import context.dispatcher
51   // Consumer coordinator started up if 'consumer' is configured
52   var coordinator: Option[ActorRef] = None
53 
54   // Consumer proxy, started up if 'consumer' is configured. Used to retrieve topic
55   // information and the data to consume
56   var consumerManager: Option[ActorRef] = None
57 
58   //Distributor actor will start up if a 'consumer' is configured
59   var distributor: Option[ActorRef] = None
60 
61   var consumerManagerHealth: Option[HealthComponent] = None
62 
63   override def preStart() = {
64     if (kafkaConfig.hasPath("producer")) {
65       startProducer()
66     }
67     if (kafkaConfig.hasPath("consumer")) {
68       startCoordinator()
69     }
70   }
71 
72   override def receive = super.receive orElse configReceive orElse {
73     // Use this call to get the Kafka Consumer Coordinator, if configured
74     case GetCoordinator =>
75       sender ! coordinator
76 
77     case GetDistributor =>
78       sender ! distributor
79 
80     case PrepareForShutdown =>
81       coordinator.foreach(_ ! PrepareForShutdown)
82       distributor.foreach(_ ! PrepareForShutdown)
83 
84     case h: HealthComponent =>
85       consumerManagerHealth = Some(h)
86   }
87 
88   def startProducer() {
89     log.info("Starting producer as wookiee-kafka config contained 'producer' config")
90     val writerCount = Try { kafkaConfig.getInt("producer-count") } getOrElse 10
91     producer = Some(context.system.actorOf(Props[KafkaWriter], "producer"))
92   }
93 
94   override def checkHealth: Future[HealthComponent] = {
95     val p = Promise[HealthComponent]()
96     val children = Seq(distributor, coordinator, producer)
97 
98     getHealth.onComplete {
99       case Success(s) =>
100         val healthFutures = children.flatten map { ref =>
101           (ref ? CheckHealth).mapTo[HealthComponent] recover {
102             case ex: Exception => HealthComponent(ref.path.name, ComponentState.CRITICAL, s"Failure to get health of child component. ${ex.getMessage}")
103           }
104         }
105 
106         Future.sequence(healthFutures) onComplete {
107           case Failure(f) =>
108             log.debug(f, "Failed to retrieve health of children objects")
109             p success HealthComponent(s.name, ComponentState.CRITICAL, s"Failure to get health of child components. ${f.getMessage}")
110           case Success(healths) =>
111             healths foreach { it => s.addComponent(it) }
112             p success s
113         }
114       case Failure(f) =>
115         log.debug(f, "Failed to get health from component")
116         p success HealthComponent(self.path.toString, ComponentState.CRITICAL, f.getMessage)
117     }
118 
119     p.future
120   }
121 
122   override protected def getHealth: Future[HealthComponent] = {
123     val h = HealthComponent(self.path.toString, ComponentState.NORMAL, "Healthy")
124     consumerManagerHealth.foreach(h.addComponent)
125     Future.successful(h)
126   }
127 
128   /**
129    * Start up all necessary actors
130    */
131   def startCoordinator() {
132     if(coordinator.isEmpty) {
133       log.info(s"Starting coordinator class")
134       consumerManager = Some(context.actorOf(KafkaTopicManager.props(), "consumer-manager"))
135       coordinator = Some(context.actorOf(Props(leader, consumerManager.get), "consumer-coordinator"))
136       distributor = Some(context.actorOf(KafkaConsumerDistributor.props(consumerManager.get), "consumer-distributor"))
137     }
138   }
139 
140   override def renewConfiguration() {
141     log.info("Received config change message, checking hosts for changes...")
142     super.renewConfiguration()
143     if (coordinator.isDefined) {
144       coordinator.get ! ConfigChange()
145     }
146   }
147 
148   override def postStop() {
149     stop
150   }
151 
152   // Stops the coordinator, unregistering this node from zookeeper
153   override def stop {
154     context.children foreach { child â‡’
155       log.info(s"Stopping child [${child.path}]")
156       context.stop(child)
157     }
158     coordinator = None
159     consumerManager = None
160     distributor = None
161     producer = None
162   }
163 }
