179    * This method is called when the system is first started and is responsible for joining an existing cluster
180    */
181   private def joinCluster: Unit = {
182     // If the cluster is not terminated and we are not a current member then join
183     if (!cluster.isTerminated && !cluster.state.members.exists(_.address == selfAddress)) {
184       createNode(membersPath, false, None)(clusterJoinTimeout) onComplete {
185         case Success(path) =>
186           getChildren(membersPath, true)(clusterJoinTimeout).mapTo[Seq[(String, Option[Array[Byte]])]] onComplete {
187             case Success(nodes) =>
188               if (nodes == Nil || nodes.size == 1) {
189                 log.info("Joining the cluster as self at {}", selfAddress)
190                 cluster.join(selfAddress)
191               }
192               else {
193                 val host = InetAddress.getLocalHost.getCanonicalHostName
194                 val protocol = selfAddress.protocol
195                 val root = selfAddress.system
196                 val local = s"${host}:${selfAddress.port.get}"
197 
198                 // Get the nodes to seed with and include our self first
199                 val count = if (nodes.length >= clusterSettings.randomSeedNodes) clusterSettings.randomSeedNodes else nodes.length
200 
201                 def getNodes(size: Int): Seq[String] = {
202                   for {
203                   // Take some extras since we are weeding out any that are not hooked to the cluster
204                     node <- Random.shuffle(nodes.take(size * 5))
205                     name = node._2 match {
206                       case Some(bytes) =>
207                         val data =  new String(bytes)
208                         parseOpt(data) match {
209                           case Some(json) =>
210                             if ((json \ "cluster-enabled").extract[Boolean])
211                               Some(node._1)
212                             else
213                               None
214                           case None => Some(node._1)
215                         }
216                       case None => Some(node._1)
217                     }
218                     if (name.isDefined)
219                   } yield name.get
220                 }
221 
222                 val sub = (Seq(local) ++ getNodes(count)).distinct.take(count)
223                 log.info(s"Seed nodes from Zookeeper are: ${sub.mkString(",")}")
224 
225                 val adds = sub.map {
226                   node =>
227                     val address = AddressFromURIString(s"${protocol}://${root}@${node}")
228 
229                     if (address.host.get.equalsIgnoreCase(host)) {
230                       // If this is this machine then the value is full qualified and we shall replace with what are akka
231                       // settings list us as
232                       Address(protocol, root, selfAddress.host.get, address.port.get)
233                     }
234                     else {
235                       address
236                     }
237                 }.toIndexedSeq
238 
239                 if (adds == Nil) {
240                   log.info("Joining the cluster as self at {}", selfAddress)
241                   cluster.join(selfAddress)
242                 }
243                 else {
244                   log.info(s"Joining the cluster with the seed nodes ${adds.mkString(",")}")
245                   cluster.joinSeedNodes(adds)
246                 }
247               }
248 
249               // Subscribe to events
250               cluster.subscribe(self, initialStateMode = InitialStateAsSnapshot,
251                 classOf[LeaderChanged], classOf[ReachabilityEvent], classOf[MemberEvent])
252 
253               // Initialize the state interval
254               stateCheck = Some(context.system.scheduler.scheduleOnce(selfStateInterval) { validateClusterMembership })
255 
256               // Send a message to the parent that we have joined the cluster
257               context.parent ! ClusterJoined
258 
259             case Failure(e) =>
260               log.error("An error occurred while trying to fetch the cluster seed nodes from Zookeeper", e)
261               context.parent ! ClusterJoinFailure
262           }
263 
264         case Failure(e) =>
265           log.error("An error occurred while trying to create the root path in Zookeeper", e)
266           context.parent ! ClusterJoinFailure
267       }
268     }
269   }

318   /**
319    * This method is called to validate our status within the clut
320    */
321   private def validateClusterMembership: Unit = {
322 
323     // The live members is the difference between all members and those that are currently unreachable
324     val live = (cluster.state.members -- cluster.state.unreachable)
325 
326     // If there were previously more then one node in the cluster we need to make sure we have not
327     // been stranded.
328     if ((cluster.state.members.size == 1 && cluster.state.members.head.address == selfAddress) ||
329       (live.size == 1 && live.head.address == selfAddress)) {
330       log.info("Verifying the cluster since we are the only node.")
331       getChildren(membersPath)(5 seconds).mapTo[Seq[(String, Option[Array[Byte]])]] onComplete {
332         case Success(nodes) =>
333           val clusterNodes = getActiveClusterNodes(nodes)
334           if (clusterNodes == Nil || clusterNodes.size == 1) {
335             // We are truly the only one so we can move on
336             log.info("We are truly the only one according to Zookeeper so we can move on")
337           }
338           else {
339             log.info(s"Cluster nodes from Zookeeper are: ${clusterNodes.mkString(",")}")
340             log.warning("I have been stranded so now we will now rejoin the cluster")
341             self ! RejoinCluster(true)
342           }
343         case Failure(e) =>
344           log.error("An error occurred while trying to fetch the cluster nodes from Zookeeper while validating membership.", e)
345           // Reschedule the check
346           stateCheck = Some(context.system.scheduler.scheduleOnce(selfStateInterval) { validateClusterMembership })
347       }
348     }
349     else {
350       // Reschedule the check
351       stateCheck = Some(context.system.scheduler.scheduleOnce(selfStateInterval) { validateClusterMembership })
352     }

355   /**
356    * This method will check the cluster's balance and is called on leadership changes. If the
357    * cluster is deemed unbalanced then a job will be scheduled to recheck it until is is balanced again.
358    */
359   private def validateCluster(retries: Int, nodesToCheck: Seq[String]): Unit = {
360     // Make sure we are the leader
361     if (leader) {
362       val nextInterval = (retries / maxRetries) * 30 * clusterStateInterval
363 
364       log.info("Verifying the cluster since we have become the leader or a previous validation failed. Checking against zookeeper")
365       getChildren(membersPath)(5 seconds).mapTo[Seq[(String, Option[Array[Byte]])]] onComplete {
366         case Success(nodes) =>
367           val clusterNodes = getActiveClusterNodes(nodes)
368 
369           if (clusterNodes != Nil && clusterNodes.size > cluster.state.members.size) {
370             // The number of cluster nodes in zookeeper does not match that of the current live cluster state
371 
372             // This may be a valid situation due to nodes that are actually shutting down properly. There might be a small window
373             // when this triggers during a node shutdown due to the timing of the nodes shutdown and removal from zookeeper. Service
374             // crashes may trigger this more often since it is unpredictable on when the cluster and zookeeper is spread around
375             // the cluster.
376             val currentMissing = clusterNodes.diff(cluster.state.members.map(m => m.address.hostPort.substring(m.address.system.length + 1)).toSeq)
377 
378             nodesToCheck.size match {
379               case 0 => // The difference is what is in zk and not in the cluster
380                 log.info("The state of cluster nodes in zookeeper shows that there are more registered then in the cluster. We will retry in a moment")
381                 balancedCluster = false
382                 context.system.scheduler.scheduleOnce(nextInterval) { validateCluster(retries - 1, currentMissing) }
383               case _ => // The missing are the nodes to check that are still missing
384                 val newNodes = nodesToCheck.diff(currentMissing)
385 
386                 newNodes.size match {
387                   case 0 => // The nodes we are watching are now ok so we can stop
388                     balancedCluster = true
389                     log.info("The cluster appears to be in order")
390                   case _ => // We still have nodes to watch so lets send them back
391                     if (retries <= 1) {
392                       balancedCluster = false
393                       log.warn("The following nodes are suspect because they are registered in Zookeeper, but are not part of the cluster: {}", newNodes.mkString(","))
394                       newNodes.foreach(n => sendRejoinMessage(AddressFromURIString(s"${selfAddress.protocol}://${selfAddress.system}@${n}")))
395                     }
396                     else {
397                       balancedCluster = false
398                       context.system.scheduler.scheduleOnce(nextInterval) { validateCluster(retries - 1, newNodes) }
399                     }
400                 }
401             }
402           }
403           else {
404             balancedCluster = true
405             log.info("The cluster appears to be in order")
406           }
407 
408         case Failure(e) =>
409           log.error("An error occurred while trying to fetch the cluster nodes from Zookeeper while validating the cluster. Retrying", e)
410           // Setup a job that check the balance again
411           context.system.scheduler.scheduleOnce(clusterStateInterval) { validateCluster(retries, nodesToCheck) }
412       }
413     }
414   }
