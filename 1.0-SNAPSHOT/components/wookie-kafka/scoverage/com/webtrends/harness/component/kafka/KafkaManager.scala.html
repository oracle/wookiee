38 /**
39  * This class manages the creation of both the KafkaConsumerCoordinator (if 'consumer' is configured)
40  * and the KafkaWriter (if 'producer' is configured). One can retrieve either of these actors by
41  * making a call to GetCoordinator or GetWriters
42  */
43 object KafkaManager {
44   // User this to retrieve the coordinator in charge of managing Workers
45   case object GetCoordinator
46   // This will return the kafka producer
47   case object GetProducer
48   //Will return the distributor
49   case object GetDistributor
50 }
51 
52 class KafkaManager(name: String) extends Component(name) with KafkaSettings {
53 
54   import KafkaManager._
55   // Kafka producer started up if 'producer' is configured
56   var producer: Option[ActorRef] = None
57 
58   // Consumer coordinator started up if 'consumer' is configured
59   var coordinator: Option[ActorRef] = None
60 
61   // Consumer proxy, started up if 'consumer' is configured. Used to retrieve topic
62   // information and the data to consume
63   var proxy: Option[ActorRef] = None
64 
65   //Distributor actor will start up if a 'consumer' is configured
66   var distributor: Option[ActorRef] = None
67 
68 
69   override def receive = super.receive orElse configReceive orElse {
70     // Use this call to get the Kafka Consumer Coordinator, if configured
71     case GetCoordinator =>
72       sender ! coordinator
73 
74     // Use this call to get the Kafka Producer, if configured
75     case GetProducer =>
76       if(producer.isEmpty)
77         log.warn("Writers not started, be sure to set 'producer' config in wookie-kafka")
78 
79       sender ! producer
80 
81     case GetDistributor =>
82       sender ! distributor
83 
84     case SystemReady =>
85       if (kafkaConfig.hasPath("producer")) {
86         startProducer()
87       }
88       if (kafkaConfig.hasPath("consumer")) {
89         startCoordinator()
90       }
91     case PrepareForShutdown =>
92       coordinator.foreach(_ ! PrepareForShutdown)
93 
94   }
95 
96   def startProducer() {
97     log.info("Starting producer as wookie-kafka config contained 'producer' config")
98     producer = Some(context.actorOf(KafkaProducer.props(), "producer"))
99   }
100 
101   /**
102    * Start up all necessary actors
103    */
104   def startCoordinator() {
105     if(coordinator.isEmpty) {
106       log.info(s"Starting coordinator class")
107       proxy = Some(context.actorOf(KafkaConsumerProxy.props(), "consumer-proxy"))
108       coordinator = Some(context.actorOf(Props(leader, proxy.get), "consumer-coordinator"))
109       distributor = Some(context.actorOf(KafkaConsumerDistributor.props(proxy.get), "consumer-distributor"))
110     }
111   }
112 
113   override def renewConfiguration() {
114     log.info("Received config change message, checking hosts for changes...")
115     super.renewConfiguration()
116     if (coordinator.isDefined) {
117       coordinator.get ! ConfigChange
118     }
119   }
120 
121   override def postStop() {
122     stop
123   }
124 
125   // Stops the coordinator, unregistering this node from zookeeper
126   override def stop {
127     context.children foreach { child â‡’
128       log.info(s"Stopping child [${child.path}]")
129       context.stop(child)
130     }
131     coordinator = None
132     proxy = None
133     distributor = None
134     producer = None
135   }
136 }
